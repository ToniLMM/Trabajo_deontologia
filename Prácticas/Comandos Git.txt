# COMANDOS DE GIT IMPORTANTES USADOS EN LA PRÁCTICA 2

# INICIALIZAR UN REPOSITORIO
git init

# CLONAR REPOSITORIO
git clone <url-repositorio-remoto>

# AÑADIR Y COMMITEAR CAMBIOS AL STAGING AREA
git add <archivo(s)>
git commit -m "<Mensaje-descriptivo>"

# CREAR UNA ETIQUETA ANNOTATED
git tag -a <nombre-tag> -m "<Mensaje-descriptivo>"

# SUBIR CAMBIOS Y TAGS AL REMOTO
git push
git push --tags

# CAMBIAR A UNA RAMA QUE SE ACABA DE CREAR
git checkout -b <nueva-rama>

# MERGEO DE RAMAS
git merge <nueva-rama>

# AÑADIR UN REPOSITORIO REMOTO A UNO LOCAL
git remote add upstream <url-repositorio-remoto>
git remote -v

# SUBIR CAMBIOS AL REPOSITORIO REMOTO ENLAZADO PREVIAMENTE
git push origin master
git push upstream master
git push --tags upstream
